# FunÃ§Ã£o para plotar elipses de confianÃ§a 95%
def plot_ellipse(ax, x, y, label, color):
    cov = np.cov(x, y)
    vals, vecs = np.linalg.eigh(cov)
    order = vals.argsort()[::-1]
    vals = vals[order]
    vecs = vecs[:, order]
    theta = np.degrees(np.arctan2(*vecs[:,0][::-1]))
    width, height = 2 * 1.96 * np.sqrt(vals)
    mean_x, mean_y = np.mean(x), np.mean(y)
    ellipse = mpatches.Ellipse((mean_x, mean_y), width, height, angle=theta,
                               alpha=0.2, color=color, label=f"95% CI â€“ {label}")
    ax.add_patch(ellipse)
